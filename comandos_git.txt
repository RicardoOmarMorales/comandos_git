1. git init //para iniciar git
2. git add . //toma una fotografía de los archivos modificados
3. reset //revierte todo lo que hace el git add
4. git commint -m
5. git checkout -- . //regresa al paso anterior aunque borre el archivo
6. git log //muestra los commit realizados
7. git commint --amend //nos permite editar los commit
8. git checkout -b rama-hereos
9. git checkout master //me paso a la rama master
10. git branch -d rama-hereos //elimina la rama hereos
11. git push
